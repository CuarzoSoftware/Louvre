908 LBox boundingBox() const noexcept
909 {
910 const LPoint &p { pos() };
911 const LPoint &s { size() };
912
913 LBox box
914 {
915 p.x(),
916 p.y(),
917 p.x() + s.w(),
918 p.y() + s.h(),
919 };
920
921 LBox childBox;
922
923 for (LView *child : children())
924 {
925 if (!child->mapped())
926 continue;
927
928 childBox = child->boundingBox();
929
930 if (childBox.x1 < box.x1)
931 box.x1 = childBox.x1;
932
933 if (childBox.y1 < box.y1)
934 box.y1 = childBox.y1;
935
936 if (childBox.x2 > box.x2)
937 box.x2 = childBox.x2;
938
939 if (childBox.y2 > box.y2)
940 box.y2 = childBox.y2;
941 }
942
943 return box;
944 }
