99 constexpr bool intersects(const LRectTemplate<N> &rect, bool inclusive = true) const noexcept
100 {
101 if(inclusive)
102 {
103 return x() <= rect.x()+rect.w() &&
104 x() + w() >= rect.x() &&
105 y() <= rect.y()+rect.h() &&
106 y()+h() >= rect.y();
107 }
108
109 return x() < rect.x()+rect.w() &&
110 x() + w() > rect.x() &&
111 y() < rect.y()+rect.h() &&
112 y()+h() > rect.y();
113 }

170 template<class N>
171 constexpr bool clip(const LRectTemplate<N> &rect) noexcept
172 {
173 T x1 = x() + w();
174 T y1 = y() + h();
175
176 T rx1 = rect.x() + rect.w();
177 T ry1 = rect.y() + rect.h();
178
179 if (x() < rect.x())
180 setX(rect.x());
181
182 if (x1 > rx1)
183 x1 = rx1;
184
185 if (y() < rect.y())
186 setY(rect.y());
187
188 if (y1 > ry1)
189 y1 = ry1;
190
191 setW(x1 - x());
192 setH(y1 - y());
193
194 if (w() < 0)
195 setW(0);
196
197 if (h() < 0)
198 setH(0);
199
200 return (w() == 0 || h() == 0);
201 }
